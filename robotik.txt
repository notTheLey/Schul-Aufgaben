class Map:
    def __init__(self, id, name, ways, size, colors, locs):
        self.id = id
        self.name = name
        self.size = size
        self.locs = locs
        self.ways = ways
        self.colors = colors

    def getID(self):
        return self.id
    
    def getName(self):
        return self.name
    
    def getSize(self):
        return self.size
    
    def getLocs(self):
        return self.locs
    
    def getWays(self):
        return self.ways
    
    def getColors(self):
        return self.colors
    
    

class Loc:
    def __init__(self, id, name, color, pos):
        self.id = id
        self.name = name
        self.color = color
        self.pos = pos

    def getID(self):
        return self.id
    
    def getName(self):
        return self.name
    
    def getColor(self):
        return self.color
    
    def getPos(self):
